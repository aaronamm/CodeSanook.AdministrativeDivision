try{
	
	var sources = Provinces.OrderBy(s => s.Id)
	.ToList()
	.Select(s => new CodeSanook_AdministrativeDivision_ProvinceRecord(){
	Id = s.Id, Code = s.Code, NameInThai = s.NameInThai, NameInEnglish = s.NameInEnglish
	});
	
	CodeSanook_AdministrativeDivision_ProvinceRecords.InsertAllOnSubmit(sources);
	
	
	SubmitChanges();
	
}catch(Exception  ex){
ex.Dump();	
	
}
